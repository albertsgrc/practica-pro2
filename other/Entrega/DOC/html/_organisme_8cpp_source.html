72  a.plantar(c, fe, fd);

120 void Organisme::crear_fill(Acel&x1, Acel&x2, Acel&fill, int&idclonsx2,
121  int&idmaxf, int&tamint, int&stams) {
122  Acel fe1, fd1, fe2, fd2;

130  /* Si els dos subarbres son no buits aleshores seguim en la interseccio,
131  * sino haurem de mirar per separat els subarbres que no siguin buits
132  * en una funcio auxiliar que es dedica a col·locar els clons */
133 
134  // Subarbres esquerres
135  Acel fef;
136  if (not fe1.es_buit() and not fe2.es_buit()) {
137  crear_fill(fe1, fe2, fef, idclonsx2, idmaxf, tamint, stams);
138  }
139  else if (not fe1.es_buit()) aux_clons_x1(fe1, fef, stams, idmaxf);
140  else if (not fe2.es_buit()) aux_clons_x2(fe2, fef, stams, idmaxf, idclonsx2);
141  // Subarbres drets
142  Acel fdf;
143  if (not fd1.es_buit() and not fd2.es_buit()) {
144  crear_fill(fd1, fd2, fdf, idclonsx2, idmaxf, tamint, stams);
145  }
146  else if (not fd1.es_buit()) aux_clons_x1(fd1, fdf, stams, idmaxf);
147  else if (not fd2.es_buit()) aux_clons_x2(fd2, fdf, stams, idmaxf, idclonsx2);
148 
149  fill.plantar(Celula(c1.consul_id(), c1.es_activa() or c2.es_activa()), fef, fdf);
150  idmaxf = max(idmaxf, c1.consul_id());
151  x1.plantar(c1, fe1, fd1);
152  x2.plantar(c2, fe2, fd2);
153 }
