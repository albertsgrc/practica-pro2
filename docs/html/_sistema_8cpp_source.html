114 void Sistema::aplicar_ronda_reproduccio(Ranking&rank) {
115  int pobaux = pob;
116  int i = 0;
117  Lemp::iterator it = empar.begin();
118  Vbool ocupat(pobaux, false);
119  nfills_ur = 0;
120  while (i < pobaux - 1 and pob < vorg.size()) {
121  if (not ocupat[i] and not vorg[i].es_mort()) {
122  bool incomp = true;
123  /* Avancem l'organisme inicial d'emparellament mentre aquest
124  * sigui mort o ja s'hi hagi emparellat. Aixo ens permet tenir
125  * una notable millora en l'eficiencia de l'operacio donat que
126  * la proxima vegada que l'organisme es reprodueixi no començarem
127  * a buscar candidats des del principi, sino des del primer id
128  * a partir del qual sabem que tots els anteriors son candidats no
129  * no valids */
130  while (incomp and (*it).ini < pobaux) {
131  int cand = (*it).ini;
132  incomp = vorg[cand].es_mort() or (*it).e[cand-i-1];
133  if (not incomp) {
134  if (not ocupat[cand]) emparellar_organismes(ocupat, rank, i, cand, it);
135  else {
136  ++cand;
137  bool emparellables = false;
138  while (not emparellables and cand < pobaux) {
139  if (not ocupat[cand] and not vorg[cand].es_mort()
140  and not (*it).e[cand-i-1]) {
141  emparellables = true;
142  emparellar_organismes(ocupat, rank, i, cand, it);
143  }
144  ++cand;
145  }
146  }
147  }
148  else ++(*it).ini;
149  }
150  }
151  ++i;
152  ++it;
153  }

Sistema::empar
Lemp empar
Llista que conté la informació dels emparellaments dels organismes existents al Sistema, excepte l&#39;últim, que pel que s&#39;ha demostrat en la descripció de la classe no en necessitem la informació dels emparellaments perquè no es podrà emparellar de moment. Té tamany pob - 1. La posició i de la llista conté un enter ini que representa l&#39;id decrementat en una unitat de l&#39;organisme amb id més petit a partir del qual l&#39;organisme amb id=i+1 pot emparellar-se en un moment donat. Donat un organisme d&#39;identificador menor a ini, o bé no s&#39;hi pot emparellar (perquè té id igual o menor), o bé és mort, o bé ja s&#39;hi ha emparellat. Si l&#39;organisme i+1 no té cap organisme per emparellar-se aleshores el valor de la posició i coincideix amb la població actual. La posició i de la llista també conté un vector de booleans e de tamany [població màxima - i - 1] (pel que es demostra en la descripció de la classe) i la posició j indica si l&#39;organisme amb id = i + 1 s&#39;ha emparellat amb l&#39;organisme amb id = i + j + 2.
