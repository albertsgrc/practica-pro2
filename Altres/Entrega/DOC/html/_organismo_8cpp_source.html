58 void Organismo::estirar_arbcelula(Arbcel&a, int&idmax) {
59  Celula c = a.arrel();
60  Arbcel fe, fd;
61  a.fills(fe, fd);
62  // Si las dos ramas son vacías fisionamos (estamos en una hoja)
63  if (fe.es_buit() and fd.es_buit()) {
64  // Creamos la célula izquierda
65  ++idmax;// Actualizamos el identificador máximo del organismo
66  Celula h1(idmax, c.es_activa());
67  Arbcel fbuit; // Creamos un árbol vacío
68  fe.plantar(h1, fbuit, fbuit);
69  // Creamos la célula derecha
70  ++idmax;
71  Celula h2(idmax, c.es_activa());
72  fd.plantar(h2, fbuit, fbuit);
73  }
74  else { // Si una de las dos ramas no es vacía entonces no hay que fisionar
75  if (not fe.es_buit()) estirar_arbcelula(fe, idmax);
76  if (not fd.es_buit()) estirar_arbcelula(fd, idmax);
77  }
78  a.plantar(c, fe, fd);
